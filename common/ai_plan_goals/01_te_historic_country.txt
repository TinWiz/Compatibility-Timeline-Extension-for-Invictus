te_great_germanisch_wanderung_aimod = {
	trigger = {
        current_date > 1128.6.28
		has_country_modifier = germanisch_wanderung
		any_neighbour_country = {
			AND = {
				NOT = {
					country_culture_group = germanic
				}
				rank = great_power
			}
		}
	}
	aggressive = 600
	migration_chance = 30
	trustworthy = -20
}

te_major_germanisch_wanderung_aimod = {
	trigger = {
        current_date > 1128.6.28
		has_country_modifier = germanisch_wanderung
		OR = {
			any_neighbour_country = {
				NOT = {
					country_culture_group = germanic
				}
				rank = major_power
			}
			any_neighbour_country = {
				count >= 2
				NOT = {
					country_culture_group = germanic
				}
				rank = great_power	
			}
		}	
	}
	aggressive = 400
	migration_chance = 20
	trustworthy = -10
}
